% mention de protection
\newcommand{\mentiondeprotection}{{%
	\bfseries\sffamily%
	\setlength{\fboxsep}{2pt}%
	\resizebox{0.4\paperwidth}{!}{%
		~\textcolor{red}{\fbox{\mention}}%
		~\hfill%
		\textcolor{blue}{\fbox{\protection}}~%
	}%
}}

% Configuration du bas de chaque transparent pour inserer la mention de
% protection si besoin
\ifmentiondeprotection
